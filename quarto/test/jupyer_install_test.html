point_selection = alt.selection_point(fields =['species'])

main_chart = (
    alt.
    Chart(temp).
    
    encode(
        x='sepal_length',
        y='sepal_width',
        color=alt.condition(
            point_selection,
            alt.Color('species:N').legend(None),
            alt.value('lightgray')
        )
    ).
    mark_point()
)

legend_chart = (
    alt.
    Chart(pd.DataFrame(temp['species'].unique(), columns = ['species'])).
    encode(
        x = alt.value(1),
        y = alt.Y('species').axis(
            titleAngle=0, 
            titleAnchor = 'start', 
            title = 'Check the circles below', 
            ticks = False, 
            labelAlign = 'left',
            labelPadding=130
        ),
        color = alt.condition(
            point_selection,
            'species:N',
            alt.value('lightgray')
        )
    ).
    mark_point()
).add_params(
    point_selection
)

result_chart = alt.hconcat(main_chart, legend_chart)
result_chart = (
    result_chart.
    configure_view(
        stroke = None
    ).
    configure_axis(
        domain = False
    )
)
